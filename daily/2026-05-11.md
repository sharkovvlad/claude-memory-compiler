**Mig 197** (utvержден):

```sql
IF v_callback = 'cmd_back' AND v_status = 'edit_phenotype' THEN
    SELECT phenotype_answers INTO v_ans_pheno FROM users WHERE telegram_id=p_telegram_id;
    v_answered := CASE
        WHEN v_ans_pheno ? 'q4' THEN 4
        WHEN v_ans_pheno ? 'q3' THEN 3
        WHEN v_ans_pheno ? 'q2' THEN 2
        WHEN v_ans_pheno ? 'q1' THEN 1
        ELSE 0
    END;
    IF v_answered = 0 THEN
        -- exit + pop_nav
        UPDATE users SET status='registered' WHERE telegram_id=p_telegram_id;
        PERFORM pop_nav(p_telegram_id);
        RETURN render_screen(p_telegram_id, 'my_plan');
    END IF;
    -- rollback last answer + render previous Q
    UPDATE users SET phenotype_answers = v_ans_pheno - ('q'||v_answered)::text WHERE telegram_id=p_telegram_id;
    IF v_answered = 1 THEN
        RETURN render_screen(p_telegram_id, 'edit_phenotype');
    ELSE
        RETURN render_screen(p_telegram_id, 'phenotype_q'||v_answered);
    END IF;
END IF;
```
